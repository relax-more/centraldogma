204     /**
205      * Starts the server. This method does nothing if the server is started already.
206      */
207     public synchronized void start() {
208         boolean success = false;
209         ThreadPoolExecutor repositoryWorker = null;
210         ProjectManager pm = null;
211         DefaultMirroringService mirroringService = null;
212         CommandExecutor executor = null;
213         Server server = null;
214         CentralDogmaSecurityManager securityManager = null;
215         try {
216             logger.info("Starting the Central Dogma ..");
217             repositoryWorker = new ThreadPoolExecutor(
218                     cfg.numRepositoryWorkers(), cfg.numRepositoryWorkers(),
219                     60, TimeUnit.SECONDS, new LinkedTransferQueue<>(),
220                     new DefaultThreadFactory("repository-worker", true));
221             repositoryWorker.allowCoreThreadTimeOut(true);
222 
223             logger.info("Starting the project manager: {}", cfg.dataDir());
224 
225             pm = new DefaultProjectManager(cfg.dataDir(), repositoryWorker, cfg.cacheSpec());
226             logger.info("Started the project manager: {}", pm);
227 
228             logger.info("Current settings:\n{}", cfg);
229 
230             mirroringService = new DefaultMirroringService(new File(cfg.dataDir(), "_mirrors"),
231                                                            pm,
232                                                            cfg.numMirroringThreads(),
233                                                            cfg.maxNumFilesPerMirror(),
234                                                            cfg.maxNumBytesPerMirror());
235 
236             if (cfg.isSecurityEnabled()) {
237                 securityManager = new CentralDogmaSecurityManager(cfg.dataDir(), securityConfig);
238             }
239 
240             logger.info("Starting the command executor ..");
241             executor = startCommandExecutor(pm, mirroringService, repositoryWorker, securityManager);
242             logger.info("Started the command executor");
243 
244             initializeInternalProject(executor);
245 
246             logger.info("Starting the RPC server");
247             server = startServer(pm, executor, securityManager);
248             logger.info("Started the RPC server at: {}", server.activePorts());
249             logger.info("Started the Central Dogma successfully");
250             success = true;
251         } finally {
252             if (success) {
253                 this.repositoryWorker = repositoryWorker;
254                 this.pm = pm;
255                 this.executor = executor;
256                 this.mirroringService = mirroringService;
257                 this.server = server;
258             } else {
259                 stop(server, executor, mirroringService, pm, repositoryWorker);
260             }
261         }
262     }

264     private CommandExecutor startCommandExecutor(
265             ProjectManager pm, DefaultMirroringService mirroringService,
266             Executor repositoryWorker, @Nullable CentralDogmaSecurityManager securityManager) {
267 
268         final CommandExecutor executor;
269         final ReplicationMethod replicationMethod = cfg.replicationConfig().method();
270         switch (replicationMethod) {
271             case ZOOKEEPER:
272                 executor = newZooKeeperCommandExecutor(pm, repositoryWorker, securityManager);
273                 break;
274             case NONE:
275                 logger.info("No replication mechanism specified; entering standalone");
276                 executor = new StandaloneCommandExecutor(pm, securityManager, repositoryWorker);
277                 break;
278             default:
279                 throw new Error("unknown replication method: " + replicationMethod);
280         }
281 
282         final CommandExecutor projInitExecutor = new ProjectInitializingCommandExecutor(executor);
283         try {
284             projInitExecutor.start(() -> {
285                 if (cfg.isMirroringEnabled()) {
286                     logger.info("Starting the mirroring service ..");
287                     mirroringService.start(projInitExecutor);
288                     logger.info("Started the mirroring service");
289                 } else {
290                     logger.info("Not starting the mirroring service because it's disabled.");
291                 }
292 
293                 if (securityManager != null) {
294                     logger.info("Starting the periodic session validation ..");
295                     securityManager.enableSessionValidation();
296                     logger.info("Started the periodic session validation");
297                 }
298             }, () -> {
299                 if (cfg.isMirroringEnabled()) {
300                     logger.info("Stopping the mirroring service ..");
301                     mirroringService.stop();
302                     logger.info("Stopped the mirroring service");
303                 }
304 
305                 if (securityManager != null) {
306                     logger.info("Stopping the periodic session validation ..");
307                     securityManager.disableSessionValidation();
308                     logger.info("Stopped the periodic session validation");
309                 }
310             });
311         } catch (Exception e) {
312             logger.warn("Failed to start the command executor. Entering read-only.", e);
313         }
314 
315         return projInitExecutor;
316     }

318     private Server startServer(ProjectManager pm, CommandExecutor executor,
319                                @Nullable CentralDogmaSecurityManager securityManager) {
320         final ServerBuilder sb = new ServerBuilder();
321         for (ServerPort p : cfg.ports()) {
322             sb.port(p);
323         }
324 
325         cfg.numWorkers().ifPresent(
326                 numWorkers -> sb.workerGroup(EventLoopGroups.newEventLoopGroup(numWorkers), true));
327         cfg.maxNumConnections().ifPresent(sb::maxNumConnections);
328         cfg.idleTimeoutMillis().ifPresent(sb::idleTimeoutMillis);
329         cfg.requestTimeoutMillis().ifPresent(sb::defaultRequestTimeoutMillis);
330         cfg.maxFrameLength().ifPresent(sb::defaultMaxRequestLength);
331         cfg.gracefulShutdownTimeout().ifPresent(
332                 t -> sb.gracefulShutdownTimeout(t.quietPeriodMillis(), t.timeoutMillis()));
333 
334         final WatchService watchService = new WatchService();
335         sb.serverListener(new ServerListenerAdapter() {
336             @Override
337             public void serverStopping(Server server) {
338                 watchService.serverStopping();
339             }
340         });
341 
342         configureThriftService(sb, pm, executor, watchService);
343 
344         sb.service("/hostname", HttpFileService.forVfs(new AbstractHttpVfs() {
345             @Override
346             public Entry get(String path, @Nullable String contentEncoding) {
347                 requireNonNull(path, "path");
348                 return new ByteArrayEntry(path, MediaType.PLAIN_TEXT_UTF_8,
349                                           server.defaultHostname().getBytes(StandardCharsets.UTF_8));
350             }
351 
352             @Override
353             public String meterTag() {
354                 return "hostname";
355             }
356         }));
357 
358         sb.service("/cache_stats", new AbstractHttpService() {
359             @Override
360             protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req)
361                     throws Exception {
362                 return HttpResponse.of(HttpStatus.OK,
363                                        MediaType.JSON_UTF_8,
364                                        Jackson.writeValueAsPrettyString(pm.cacheStats()));
365             }
366         });
367 
368         sb.service("/security_enabled", new AbstractHttpService() {
369             @Override
370             protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req)
371                     throws Exception {
372                 return HttpResponse.of(HttpStatus.OK,
373                                        MediaType.JSON_UTF_8,
374                                        Jackson.writeValueAsPrettyString(cfg.isSecurityEnabled()));
375             }
376         });
377 
378         sb.service("/monitor/l7check", new HttpHealthCheckService());
379 
380         // TODO(hyangtack): This service is temporarily added to support redirection from '/docs' to '/docs/'.
381         //                  It would be removed if this kind of redirection is handled by Armeria.
382         sb.service("/docs", new AbstractHttpService() {
383             @Override
384             protected HttpResponse doGet(ServiceRequestContext ctx, HttpRequest req)
385                     throws Exception {
386                 return HttpResponse.of(HttpHeaders.of(HttpStatus.TEMPORARY_REDIRECT)
387                                                   .set(HttpHeaderNames.LOCATION, "/docs/"));
388             }
389         });
390         sb.serviceUnder("/docs/",
391                         new DocServiceBuilder().exampleHttpHeaders(
392                                 CentralDogmaService.class,
393                                 HttpHeaders.of(HttpHeaderNames.AUTHORIZATION,
394                                                "bearer " + CsrfToken.ANONYMOUS))
395                                                .build());
396 
397         configureHttpApi(sb, pm, executor, watchService, securityManager);
398 
399         final Server s = sb.build();
400         s.start().join();
401         return s;
402     }

404     private CommandExecutor newZooKeeperCommandExecutor(ProjectManager pm, Executor repositoryWorker,
405                                                         @Nullable CentralDogmaSecurityManager securityManager) {
406 
407         final ZooKeeperReplicationConfig zkCfg = (ZooKeeperReplicationConfig) cfg.replicationConfig();
408 
409         // Read or generate the replica ID.
410         final File replicaIdFile =
411                 new File(cfg.dataDir().getAbsolutePath() + File.separatorChar + "replica_id");
412         final String replicaId;
413 
414         if (replicaIdFile.exists()) {
415             // Read the replica ID.
416             try {
417                 final List<String> lines = Files.readAllLines(replicaIdFile.toPath());
418                 if (lines.isEmpty()) {
419                     throw new IllegalStateException("replica_id contains no lines.");
420                 }
421                 replicaId = lines.get(0).trim();
422                 if (replicaId.isEmpty()) {
423                     throw new IllegalStateException("replica_id is empty.");
424                 }
425             } catch (Exception e) {
426                 throw new ReplicationException("failed to retrieve the replica ID from: " + replicaIdFile, e);
427             }
428             logger.info("Using ZooKeeper-based replication mechanism with an existing replica ID: {}",
429                         replicaId);
430         } else {
431             // Generate a replica ID.
432             replicaId = UUID.randomUUID().toString();
433             try {
434                 Files.write(replicaIdFile.toPath(), ImmutableList.of(replicaId));
435             } catch (Exception e) {
436                 throw new ReplicationException("failed to generate a replica ID into: " + replicaIdFile, e);
437             }
438             logger.info("Using ZooKeeper-based replication mechanism with a generated replica ID: {}",
439                         replicaId);
440         }
441 
442         // TODO(trustin): Provide a way to restart/reload the replicator
443         //                so that we can recover from ZooKeeper maintenance automatically.
444         final File revisionFile =
445                 new File(cfg.dataDir().getAbsolutePath() + File.separatorChar + "last_revision");
446 
447         return ZooKeeperCommandExecutor.builder()
448                                        .replicaId(replicaId)
449                                        .delegate(new StandaloneCommandExecutor(replicaId, pm,
450                                                                                securityManager,
451                                                                                repositoryWorker))
452                                        .numWorkers(zkCfg.numWorkers())
453                                        .connectionString(zkCfg.connectionString())
454                                        .timeoutMillis(zkCfg.timeoutMillis())
455                                        .createPathIfNotExist(true)
456                                        .path(zkCfg.pathPrefix())
457                                        .maxLogCount(zkCfg.maxLogCount())
458                                        .minLogAge(zkCfg.minLogAgeMillis(), TimeUnit.MILLISECONDS)
459                                        .revisionFile(revisionFile)
460                                        .build();
461     }

463     private void configureThriftService(ServerBuilder sb, ProjectManager pm, CommandExecutor executor,
464                                         WatchService watchService) {
465         final CentralDogmaServiceImpl service =
466                 new CentralDogmaServiceImpl(pm, executor, watchService);
467 
468         Service<HttpRequest, HttpResponse> thriftService =
469                 ThriftCallService.of(service)
470                                  .decorate(CentralDogmaTimeoutScheduler::new)
471                                  .decorate(CentralDogmaExceptionTranslator::new)
472                                  .decorate(THttpService.newDecorator());
473 
474         if (cfg.isCsrfTokenRequiredForThrift()) {
475             thriftService = thriftService.decorate(HttpAuthService.newDecorator(new CsrfTokenAuthorizer()));
476         } else {
477             thriftService = thriftService.decorate(TokenlessClientLogger::new);
478         }
479 
480         sb.service("/cd/thrift/v1", thriftService);
481     }

483     private void configureHttpApi(ServerBuilder sb,
484                                   ProjectManager pm, CommandExecutor executor, WatchService watchService,
485                                   @Nullable CentralDogmaSecurityManager securityManager) {
486         final String apiV0PathPrefix = "/api/v0/";
487 
488         final TokenService tokenService = new TokenService(pm, executor);
489         final MetadataService mds = new MetadataService(pm, executor);
490 
491         final Function<Service<HttpRequest, HttpResponse>,
492                 ? extends Service<HttpRequest, HttpResponse>> decorator;
493         if (cfg.isSecurityEnabled()) {
494             requireNonNull(securityManager, "securityManager");
495             sb.service(apiV0PathPrefix + "authenticate", new LoginService(securityManager, executor))
496               .service(apiV0PathPrefix + "logout", new LogoutService(securityManager, executor));
497 
498             final ApplicationTokenAuthorizer ata = new ApplicationTokenAuthorizer(tokenService::findToken);
499             final SessionTokenAuthorizer sta = new SessionTokenAuthorizer(securityManager);
500 
501             decorator = HttpAuthService.newDecorator(ata, sta);
502         } else {
503             decorator = HttpAuthService.newDecorator(new CsrfTokenAuthorizer());
504         }
505 
506         final SafeProjectManager safePm = new SafeProjectManager(pm);
507 
508         final HttpApiRequestConverter v1RequestConverter = new HttpApiRequestConverter(safePm);
509         final HttpApiResponseConverter v1ResponseConverter = new HttpApiResponseConverter();
510 
511         sb.annotatedService(API_V1_PATH_PREFIX,
512                             new ProjectServiceV1(safePm, executor, mds), decorator,
513                             v1RequestConverter, v1ResponseConverter);
514         sb.annotatedService(API_V1_PATH_PREFIX,
515                             new RepositoryServiceV1(safePm, executor, mds), decorator,
516                             v1RequestConverter, v1ResponseConverter);
517         sb.annotatedService(API_V1_PATH_PREFIX,
518                             new ContentServiceV1(safePm, executor, watchService), decorator,
519                             v1RequestConverter, v1ResponseConverter);
520 
521         if (cfg.isWebAppEnabled()) {
522             final RestfulJsonResponseConverter httpApiV0Converter = new RestfulJsonResponseConverter();
523 
524             // TODO(hyangtack): Simplify this if https://github.com/line/armeria/issues/582 is resolved.
525             sb.annotatedService(apiV0PathPrefix, new UserService(safePm, executor),
526                                 decorator, httpApiV0Converter)
527               .annotatedService(apiV0PathPrefix, new ProjectService(safePm, executor),
528                                 decorator, httpApiV0Converter)
529               .annotatedService(apiV0PathPrefix, new RepositoryService(safePm, executor),
530                                 decorator, httpApiV0Converter)
531               .annotatedService(apiV0PathPrefix, tokenService, decorator, httpApiV0Converter);
532         }
533 
534         sb.serviceUnder("/", HttpFileService.forClassPath("webapp"));
535     }

537     /**
538      * Stops the server. This method does nothing if the server is stopped already.
539      */
540     public synchronized void stop() {
541         if (server == null) {
542             return;
543         }
544 
545         final Server server = this.server;
546         final CommandExecutor executor = this.executor;
547         final DefaultMirroringService mirroringService = this.mirroringService;
548         final ProjectManager pm = this.pm;
549         final ExecutorService repositoryWorker = this.repositoryWorker;
550 
551         this.server = null;
552         this.executor = null;
553         this.mirroringService = null;
554         this.pm = null;
555         this.repositoryWorker = null;
556 
557         logger.info("Stopping the Central Dogma ..");
558         if (!stop(server, executor, mirroringService, pm, repositoryWorker)) {
559             logger.warn("Stopped the Central Dogma with failure");
560         } else {
561             logger.info("Stopped the Central Dogma successfully");
562         }
563     }

565     private static boolean stop(
566             Server server, CommandExecutor executor, DefaultMirroringService mirroringService,
567             ProjectManager pm, ExecutorService repositoryWorker) {
568 
569         boolean success = true;
570         try {
571             if (server != null) {
572                 logger.info("Stopping the RPC server ..");
573                 server.stop().join();
574                 logger.info("Stopped the RPC server");
575             }
576         } catch (Exception e) {
577             success = false;
578             logger.warn("Failed to stop the RPC server:", e);
579         } finally {
580             try {
581                 if (executor != null) {
582                     logger.info("Stopping the command executor ..");
583                     executor.stop();
584                     logger.info("Stopped the command executor");
585                 }
586             } catch (Exception e) {
587                 success = false;
588                 logger.warn("Failed to stop the command executor:", e);
589             } finally {
590                 try {
591                     // Stop the mirroring service if the command executor did not stop it.
592                     if (mirroringService != null && mirroringService.isStarted()) {
593                         logger.info("Stopping the mirroring service not terminated by the command executor ..");
594                         mirroringService.stop();
595                         logger.info("Stopped the mirroring service");
596                     }
597                 } catch (Exception e) {
598                     success = false;
599                     logger.warn("Failed to stop the mirroring service:", e);
600                 } finally {
601                     try {
602                         if (pm != null) {
603                             logger.info("Stopping the project manager ..");
604                             pm.close();
605                             logger.info("Stopped the project manager");
606                         }
607                     } catch (Exception e) {
608                         success = false;
609                         logger.warn("Failed to stop the project manager:", e);
610                     } finally {
611                         if (repositoryWorker != null && !repositoryWorker.isTerminated()) {
612                             logger.info("Stopping the repository worker ..");
613                             boolean interruptLater = false;
614                             while (!repositoryWorker.isTerminated()) {
615                                 repositoryWorker.shutdownNow();
616                                 try {

621                                 }
622                             }
