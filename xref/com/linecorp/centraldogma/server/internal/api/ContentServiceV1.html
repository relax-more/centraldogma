90      /**
91       * GET /projects/{projectName}/repos/{repoName}/tree{path}?revision={revision}
92       *
93       * <p>Returns the list of files in the path.
94       */
95      @Get("regex:/projects/(?<projectName>[^/]+)/repos/(?<repoName>[^/]+)/tree(?<path>(|/.*))$")
96      public CompletionStage<List<EntryDto<?>>> listFiles(@Param("path") String path,
97                                                          @Param("revision") @Default("-1") String revision,
98                                                          @RequestObject Repository repository) {
99          final String path0 = rootDirIfEmpty(path);
100         return listFiles(repository, path0, new Revision(revision),
101                          ImmutableMap.of(FindOption.FETCH_CONTENT, false));
102     }
103 
104     private static CompletionStage<List<EntryDto<?>>> listFiles(Repository repository, String path,
105                                                                 Revision revision,
106                                                                 Map<FindOption<?>, ?> options) {
107         final String pathPattern = appendWildCardIfDirectory(path);
108         return repository.find(revision, pathPattern, options)
109                          .thenApply(entries -> entries.values().stream()
110                                                       .filter(entry -> entry.type() != EntryType.DIRECTORY)
111                                                       .map(DtoConverter::convert).collect(toImmutableList()));
112     }
113 
114     private static String appendWildCardIfDirectory(String path) {
115         return path.endsWith("/") ? path + '*' : path;
116     }
117 
118     /**
119      * Returns "/" if the specified path is null or empty.
120      */
121     private static String rootDirIfEmpty(String path) {
122         return isNullOrEmpty(path) ? "/" : path;
123     }

125     /**
126      * POST /projects/{projectName}/repos/{repoName}/contents?revision={revision}
127      *
128      * <p>Adds or edits a file.
129      */
130     @Post("/projects/{projectName}/repos/{repoName}/contents")
131     public CompletionStage<?> commit(@Param("revision") @Default("-1") String revision,
132                                      @RequestObject Repository repository,
133                                      @RequestObject Author author,
134                                      @RequestObject CommitMessageDto commitMessage,
135                                      @RequestObject(ChangesRequestConverter.class)
136                                              Iterable<Change<?>> changes) {
137         final Revision normalizedRevision = repository.normalizeNow(new Revision(revision));
138 
139         final CompletableFuture<Map<String, Change<?>>> changesFuture =
140                 repository.previewDiff(normalizedRevision, changes);
141 
142         return changesFuture.thenCompose(previewDiffs -> {
143             final long commitTimeMillis = System.currentTimeMillis();
144             if (previewDiffs.size() == 0) {
145                 throw new RedundantChangeException();
146             }
147             final CompletableFuture<Revision> resultRevisionFuture =
148                     push(commitTimeMillis, author, repository, normalizedRevision,
149                          commitMessage, previewDiffs.values()).toCompletableFuture();
150             final String pathPattern = joinPaths(changes);
151             final CompletableFuture<Map<String, Entry<?>>> findFuture = resultRevisionFuture.thenCompose(
152                     result -> repository.find(result, pathPattern));
153 
154             return findFuture.thenApply(entries -> objectOrList(
155                     entries.values(), Iterables.size(changes) != 1, (collections) -> convertEntry(
156                             resultRevisionFuture.join(), // resultRevisionFuture is already complete
157                             repository, collections, commitTimeMillis)));
158         });
159     }
160 
161     private CompletionStage<Revision> push(long commitTimeMills, Author author, Repository repository,
162                                            Revision revision, CommitMessageDto commitMessage,
163                                            Iterable<Change<?>> changes) {
164         final String summary = commitMessage.summary();
165         final String detail = commitMessage.detail();
166         final Markup markup = commitMessage.markup();
167 
168         return execute(Command.push(
169                 commitTimeMills, author, repository.parent().name(), repository.name(),
170                 revision, summary, detail, markup, changes));
171     }
172 
173     private static String joinPaths(Iterable<Change<?>> changes) {
174         final StringBuilder sb = new StringBuilder();
175         for (Change c : changes) {
176             if (c.type() == ChangeType.RENAME) {
177                 sb.append(c.contentAsText()); // newPath
178             } else {
179                 sb.append(c.path());
180             }
181             sb.append(',');
182         }
183         return sb.toString();
184     }

186     private static EntryDto<?> convertEntry(Revision revision, Repository repository,
187                                             Entry<?> entry, long commitTimeMillis) {
188         return DtoConverter.convert(revision, repository.parent().name(), repository.name(),
189                                     entry.path(), entry.type(), commitTimeMillis);
190     }

192     /**
193      * GET /projects/{projectName}/repos/{repoName}/contents{path}?revision={revision}&amp;
194      * jsonpath={jsonpath}
195      *
196      * <p>Returns the entry of files in the path. This is same with
197      * {@link #listFiles(String, String, Repository)} except that containing the content of the files.
198      * Note that if the {@link HttpHeaderNames#IF_NONE_MATCH} in which has a revision is sent with,
199      * this will await for the time specified in {@link HttpHeaderNames#PREFER}.
200      * During the time if the specified revision becomes different with the latest revision, this will
201      * response back right away to the client with the {@link WatchResultDto}.
202      * {@link HttpStatus#NOT_MODIFIED} otherwise.
203      */
204     @Get("regex:/projects/(?<projectName>[^/]+)/repos/(?<repoName>[^/]+)/contents(?<path>(|/.*))$")
205     public CompletionStage<?> getFiles(@Param("path") String path,
206                                        @Param("revision") @Default("-1") String revision,
207                                        @RequestObject Repository repository,
208                                        @RequestObject(WatchRequestConverter.class)
209                                                Optional<WatchRequest> watchRequest,
210                                        @RequestObject(QueryRequestConverter.class) Optional<Query<?>> query) {
211         final String path0 = rootDirIfEmpty(path);
212 
213         // watch repository or a file
214         if (watchRequest.isPresent()) {
215             final Revision lastKnownRevision = watchRequest.get().lastKnownRevision();
216             long timeOutMillis = watchRequest.get().timeoutMillis();
217             if (query.isPresent()) {
218                 return watchFile(repository, lastKnownRevision, query.get(), timeOutMillis);
219             }
220 
221             return watchRepository(repository, lastKnownRevision, path0, timeOutMillis);
222         }
223 
224         if (query.isPresent()) {
225             // get a file
226             return repository.get(new Revision(revision), query.get())
227                              .handle(returnOrThrow((QueryResult<?> result) ->
228                                                            DtoConverter.convert(result, path0)));
229         }
230 
231         // get files
232         return listFiles(repository, path0, new Revision(revision), ImmutableMap.of());
233     }
234 
235     private CompletionStage<?> watchFile(Repository repository, Revision lastKnownRevision,
236                                          Query<?> query, long timeOutMillis) {
237         final CompletableFuture<? extends QueryResult<?>> future = watchService.watchFile(
238                 repository, lastKnownRevision, query, timeOutMillis);
239 
240         return future.thenCompose(result -> handleWatchSuccess(repository, result.revision(), query.path()))
241                      .exceptionally(this::handleWatchFailure);
242     }
243 
244     private static CompletionStage<Object> handleWatchSuccess(Repository repository,
245                                                               Revision revision, String pathPattern) {
246         final CompletableFuture<List<Commit>> historyFuture =
247                 repository.history(revision, revision, pathPattern);
248         return repository.find(revision, pathPattern)
249                          .thenCombine(historyFuture, (entryMap, commits) -> {
250                              // the size of commits should be 1
251                              return DtoConverter.convert(commits.get(0), entryMap.values(),
252                                                          repository.parent().name(), repository.name());
253                          });
254     }
255 
256     private Object handleWatchFailure(Throwable thrown) {
257         if (Throwables.getRootCause(thrown) instanceof CancellationException &&
258             !watchService.isServerStopping()) {
259 
260             // timeout happens
261             return HttpResponse.of(HttpStatus.NOT_MODIFIED);
262         }
263         return Exceptions.throwUnsafely(thrown);
264     }
265 
266     private CompletionStage<?> watchRepository(Repository repository, Revision lastKnownRevision,
267                                                String path, long timeOutMillis) {
268         final String pathPattern = appendWildCardIfDirectory(path);
269         final CompletableFuture<Revision> future =
270                 watchService.watchRepository(repository, lastKnownRevision, pathPattern, timeOutMillis);
271 
272         return future.thenCompose(revision -> handleWatchSuccess(repository, revision, pathPattern))
273                      .exceptionally(this::handleWatchFailure);
274     }

276     /**
277      * GET /projects/{projectName}/repos/{repoName}/commits/{revision}?path={path}&amp;to={to}
278      *
279      * <p>Returns a commit or the list of commits in the path. If the user specify the {@code revision} only,
280      * this will return the corresponding commit. If the user does not specify the {@code revision} or
281      * specify {@code to}, this will return the list of commits.
282      */
283     @Get("regex:/projects/(?<projectName>[^/]+)/repos/(?<repoName>[^/]+)/commits(?<revision>(|/.*))$")
284     public CompletionStage<?> listCommits(@Param("revision") String revision,
285                                           @Param("path") @Default("/**") String path,
286                                           @Param("to") Optional<String> to,
287                                           @RequestObject Repository repository) {
288         final Revision fromRevision;
289         final Revision toRevision;
290 
291         // 1. only the "revision" is specified:       get the "revision" and return just one commit
292         // 2. only the "to" is specified:             get from "HEAD" to "to" and return the list
293         // 3. the "revision" and "to" is specified:   get from the "revision" to "to" and return the list
294         // 4. nothing is specified:                   get from "HEAD" to "INIT" and return the list
295         if (isNullOrEmpty(revision) || "/".equalsIgnoreCase(revision)) {
296             fromRevision = Revision.HEAD;
297             toRevision = to.map(Revision::new).orElse(Revision.INIT);
298         } else {
299             fromRevision = new Revision(revision.substring(1));
300             toRevision = to.map(Revision::new).orElse(fromRevision);
301         }
302 
303         final RevisionRange range = repository.normalizeNow(fromRevision, toRevision);
304 
305         return repository
306                 .history(range.from(), range.to(), path)
307                 .thenApply(commits -> {
308                     final boolean toList = isNullOrEmpty(revision) || "/".equalsIgnoreCase(revision) ||
309                                            to.isPresent();
310                     return objectOrList(commits, toList, DtoConverter::convert);
311                 });
312     }

314     /**
315      * GET /projects/{projectName}/repos/{repoName}/compare?
316      * path={path}&amp;from={from}&amp;to={to}&amp;jsonpath={jsonpath}
317      *
318      * <p>Returns the diffs.
319      */
320     @Get("/projects/{projectName}/repos/{repoName}/compare")
321     public CompletionStage<?> getDiff(@Param("path") @Default("/**") String path,
322                                       @Param("from") @Default("-1") String from,
323                                       @Param("to") @Default("1") String to,
324                                       @RequestObject Repository repository,
325                                       @RequestObject(QueryRequestConverter.class) Optional<Query<?>> query) {
326         if (query.isPresent()) {
327             return repository.diff(new Revision(from), new Revision(to), query.get())
328                              .thenApply(DtoConverter::convert);
329         }
330         return repository
331                 .diff(new Revision(from), new Revision(to), path)
332                 .thenApply(changeMap -> objectOrList(changeMap.values(), true, DtoConverter::convert));
333     }
334 
335     private static <T> Object objectOrList(Collection<T> collection, boolean toList, Function<T, ?> converter) {
336         if (collection.isEmpty()) {
337             return ImmutableList.of();
338         }
339         if (toList) {
340             return collection.stream().map(converter).collect(toImmutableList());
341         }
342         return converter.apply(Iterables.getOnlyElement(collection));
343     }
