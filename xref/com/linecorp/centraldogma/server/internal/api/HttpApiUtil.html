43  /**
44   * A utility class which provides common functions for HTTP API.
45   */
46  final class HttpApiUtil {
47  
48      static final JsonNode unremoveRequest = Jackson.valueToTree(
49              ImmutableList.of(
50                      ImmutableMap.of("op", "replace",
51                                      "path", "/status",
52                                      "value", "active")));
53  
54      /**
55       * Returns a newly created {@link HttpResponseException} with the specified {@link HttpStatus} and
56       * {@code message}.
57       */
58      static HttpResponseException newHttpResponseException(HttpStatus status, String message) {
59          return HttpResponseException.of(newResponseWithErrorMessage(status, message));
60      }
61  
62      /**
63       * Returns a newly created {@link HttpResponse} with the specified {@link HttpStatus} and {@code message}.
64       */
65      static HttpResponse newResponseWithErrorMessage(HttpStatus status, String message) {
66          // TODO(minwoox) refine the error message
67          final ObjectNode content = JsonNodeFactory.instance.objectNode().put("message", message);
68          try {
69              return HttpResponse.of(status, MediaType.JSON_UTF_8, Jackson.writeValueAsBytes(content));
70          } catch (JsonProcessingException e) {
71              // should not reach here
72              throw new Error(e);
73          }
74      }
75  
76      /**
77       * Ensures the specified {@link JsonNode} equals to {@link HttpApiUtil#unremoveRequest}.
78       *
79       * @throws IllegalArgumentException if the {@code node} does not equal to
80       *                                  {@link HttpApiUtil#unremoveRequest}.
81       */
82      static void checkUnremoveArgument(JsonNode node) {
83          checkArgument(unremoveRequest.equals(node),
84                        "Unsupported JSON patch: " + node +
85                        " (expected: " + unremoveRequest + ')');
86      }
87  
88      /**
89       * Ensures the specified {@code status} is valid.
90       *
91       * @throws IllegalArgumentException if the {@code status} is invalid.
92       */
93      static void checkStatusArgument(String status) {
94          Preconditions.checkArgument("removed".equalsIgnoreCase(status),
95                                      "invalid status: " + status + " (expected: removed)");
96      }
97  
98      /**
99       * Throws the specified {@link Throwable} if it is not {@code null}. This is a special form to be used
100      * for {@link CompletionStage#handle(BiFunction)}.
101      */
102     @SuppressWarnings("unused")
103     static Void throwUnsafelyIfNonNull(@Nullable Object unused,
104                                        @Nullable Throwable cause) {
105         throwUnsafelyIfNonNull(cause);
106         return null;
107     }
108 
109     /**
110      * Throws the specified {@link Throwable} if it is not {@code null}.
111      */
112     static void throwUnsafelyIfNonNull(@Nullable Throwable cause) {
113         if (cause != null) {
114             Exceptions.throwUnsafely(cause);
115         }
116     }
117 
118     /**
119      * Returns a {@link BiFunction} for a {@link CompletionStage#handle(BiFunction)} which returns an object
120      * if the specified {@link Throwable} is {@code null}. Otherwise, it throws the {@code cause}.
121      */
122     @SuppressWarnings("unchecked")
123     static <T> BiFunction<? super Void, Throwable, ? extends T> returnOrThrow(Supplier<? super T> supplier) {
124         return (unused, cause) -> {
125             throwUnsafelyIfNonNull(cause);
126             return (T) supplier.get();
127         };
128     }
129 
130     /**
131      * Returns a {@link BiFunction} for a {@link CompletionStage#handle(BiFunction)} which returns an object
132      * applied by the specified {@code function} if the specified {@link Throwable} is {@code null}.
133      * Otherwise, it throws the {@code cause}.
134      */
135     @SuppressWarnings("unchecked")
136     static <T, U> BiFunction<? super T, Throwable, ? extends U> returnOrThrow(
137             Function<? super T, ? extends U> function) {
138         return (result, cause) -> {
139             throwUnsafelyIfNonNull(cause);
140             return function.apply(result);
141         };
142     }
143 
144     private HttpApiUtil() {}
145 }
